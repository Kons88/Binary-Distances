## Example Matrix:
w <- c(1,0,0,1,1,0,1,1,1)
x <- c(0,1,0,0,0,1,0,0,1)
y <- c(0,0,1,0,0,0,0,0,1)
z <- c(1,1,0,0,0,1,0,0,0)



m <- cbind(w,x,y,z)
m


binary.dist <- function(m, method = "jaccard", row.dist = FALSE, diag = FALSE, upper = FALSE){
  ##----------------------------------------------------------------------------------
  ## m is a binary matrix, or a data-frame of 0s and 1s.
  ##
  ## The method argument takes the values: 
  ## "jaccard" (or "tanimoto"), "sorensen-dice (or "czekanowski", or "nei-li"), 
  ## "3w-jaccard","simple matching" (or "sokal-michener"), "sokal-sneath 1", 
  ## "sokal-sneath 2", "sokal-sneath 3", "sokal-sneath 4", "sokal-sneath 5" 
  ## (or "ochiai 2"), "rogers-tanimoto", "faith", "gower-legendre", "russel-rao",
  ## "ochiai 1" (or "cosine", or "otsuka-ochiai", or "fowlkes-mallows index"),
  ## "forbes 1", "forbes 2", "fossum", "sorgenfrei", "mountford", "mcconnaughey",
  ## "tarwid", "kulczynski 1", "kulczynski 2" (or "driver-kroeber"), "johnson",
  ## "dennis", "simpson" (or "szymkiewicz-simpson"), "braun-blanquet", "fager-mcgowan",
  ## "gower", "pearson 1", "pearson 2", "pearson 3", "pearson-heron 1", "pearson-heron 2"
  ## "cole", "stiles", "yule's q", "yule's w" (or "yule's y"), "dispersion", "hamann",
  ## "mcewen-michael", "goodman-kruskal lambda", "anderberg", "baroni-urbani-buser 1",
  ## "baroni-urbani-buser 2", "peirce", "eyraud" and "tarantula". 
  ## Its default value is "jaccard".
  ##
  ## row.dist represents the way one needs to calculate the distances. If it is
  ## TRUE, then distances between the rows are calculated according to the method 
  ## chosen. And if it is FALSE, then the distances between each column is calculated.
  ##----------------------------------------------------------------------------------
  if(is.matrix(m)==FALSE & is.data.frame(m)==FALSE){
     warning("ERROR: The object you are trying to use is not a matrix or a data-frame.")
  }else{ 

   if(is.data.frame(m)==TRUE){m <- as.matrix(m)}

   if(row.dist==FALSE){

     dist.mat <- matrix(0, nrow=ncol(m), ncol=ncol(m)) # pre-allocate the distance matrix.
     colnames(dist.mat) <- colnames(m)
     rownames(dist.mat) <- colnames(m)
  
     ######## The Jaccard (or Tanimoto) Distance ########
      if(method == "Jaccard" || method == "Jaccar" || method == "Jacca" || method == "Jacc" || method == "Jac" || method == "Tanimoto" || method == "Tanimot" || method == "Tanimo" || method == "Tanim" || method == "Tani" || method == "Tan" || method == "Ta"||method == "jaccard"||method=="jaccar"||method=="jacca"||method=="jacc"||method=="jac"||method=="tanimoto"||method=="tanimot"||method=="tanimo"||method=="tanim"||method=="tani"||method=="tan"||method=="ta"||method==1){
       for(i in 1:ncol(m)){
        for(j in 1:ncol(m)){
           dist.mat[i,j] <- 1-(table(m[,i],m[,j])[2,2]/(table(m[,i],m[,j])[1,2]+table(m[,i],m[,j])[2,1]+table(m[,i],m[,j])[2,2]))
        }
       }
      }

     ######## The Sorensen-Dice (or Czekanowski or Nei & Li) Distance ########
      if(method == "SD"||method == "Sorensen-Dice"||method == "Sorensen-Dic"||method == "Sorensen-Di"||method == "Sorensen-Di"||method == "Sorensen-D"||method == "Sorensen-"||method == "Sorensen"||method == "Sorense"||method == "Sorens"||method == "Soren"||method == "Dice"||method == "Dic"||method == "Czekanowski"||method == "Czekanowsk"||method == "Czekanows"||method == "Czekanow"||method == "Czekano"||method == "Czekan"||method == "Czeka"||method == "Czek"||method == "Cze"||method == "Nei-Li"||method == "Nei-L"||method == "Nei-"||method == "NL"||method=="sd"||method=="sorensen-dice"||method=="sorensen-dic"||method=="sorensen-di"||method=="sorensen-di"||method=="sorensen-d"||method=="sorensen-"||method=="sorensen"||method=="sorense"||method=="sorens"||method=="soren"||method=="dice"||method=="dic"||method=="czekanowski"||method=="czekanowsk"||method=="czekanows"||method=="czekanow"||method=="czekano"||method=="czekan"||method=="czeka"||method=="czek"||method=="cze"||method=="nei-li"||method=="nei-l"||method=="nei-"||method=="nl"||method==2){
       for(i in 1:ncol(m)){
        for(j in 1:ncol(m)){
           dist.mat[i,j] <- 1-((2*table(m[,i],m[,j])[2,2])/(table(m[,i],m[,j])[1,2]+table(m[,i],m[,j])[2,1]+2*table(m[,i],m[,j])[2,2]))
        }
       }
      }

     ######## The 3W-Jaccard Distance ########
      if(method == "3W-Jaccard"||method == "3W-Jaccar"||method == "3W-Jacca"||method == "3W-Jacc"||method == "3W-Jac"||method == "3W-Ja"||method == "3W-J"||method=="3W"||method=="3WJaccard"||method=="3WJaccar"||method=="3WJacca"||method=="3WJacc"||method=="3WJ"||method == "3w-jaccard"||method == "3w-jaccar"||method == "3w-jacca"||method == "3w-jacc"||method == "3w-jac"||method == "3w-ja"||method == "3w-j"||method=="3w"||method=="3wjaccard"||method=="3wjaccar"||method=="3wjacca"||method=="3wjacc"||method=="3wj"||method==3){
       for(i in 1:ncol(m)){
        for(j in 1:ncol(m)){
           dist.mat[i,j] <- 1-((3*table(m[,i],m[,j])[2,2])/(table(m[,i],m[,j])[1,2]+table(m[,i],m[,j])[2,1]+3*table(m[,i],m[,j])[2,2]))
        }
       }
      }

     ######## The Simple Matching (or Sokal & Michener) Distance ########
      if(method == "Simple Matching"||method == "Simple Matchin"||method == "Simple Matchi"||method == "Simple Match"||method == "Simple Matc"||method == "Simple Mat"||method == "Simple Ma"||method == "Simple M"||method == "Simple"||method == "Simpl"||method == "Simp"||method == "Sim"||method=="Sokal-Michener"||method=="Sokal-Michene"||method=="Sokal-Michen"||method=="Sokal-Miche"||method=="Sokal-Mich"||method=="Sokal-Mic"||method=="Sokal-Mi"||method=="Sokal-M"||method=="Sokal-M"||method=="SM"||method == "simple matching"||method == "simple matchin"||method == "simple matchi"||method == "simple match"||method == "simple matc"||method == "simple mat"||method == "simple ma"||method == "simple m"||method == "simple"||method == "simpl"||method == "simp"||method == "sim"||method=="sokal-michener"||method=="sokal-michene"||method=="sokal-michen"||method=="sokal-miche"||method=="sokal-mich"||method=="sokal-mic"||method=="sokal-mi"||method=="sokal-m"||method=="sokal-m"||method=="sm"||method==4){
       for(i in 1:ncol(m)){
        for(j in 1:ncol(m)){
           dist.mat[i,j] <- 1-((table(m[,i],m[,j])[2,2]+table(m[,i],m[,j])[1,1])/(table(m[,i],m[,j])[1,2]+table(m[,i],m[,j])[2,1]+table(m[,i],m[,j])[2,2]+table(m[,i],m[,j])[1,1]))
        }
       }
      }

     ######## The Sokal & Sneath I Distance ########
      if(method == "Sokal-Sneath 1"||method=="Sokal-Sneat 1"||method=="Sokal-Snea 1"||method=="Sokal-Sne 1"||method=="Sokal-Sn 1"||method=="Sokal-S 1"||method=="SS1"||method=="SS 1"||method=="Sokal-Sneath I"||method=="Sokal-Sneat I"||method=="Sokal-Snea I"||method=="Sokal-Sne I"||method=="Sokal-Sn I"||method=="Sokal-S I"||method=="SSI"||method=="SS I"||method == "sokal-sneath 1"||method=="sokal-sneat 1"||method=="sokal-snea 1"||method=="sokal-sne 1"||method=="sokal-sn 1"||method=="sokal-s 1"||method=="ss1"||method=="ss 1"||method=="sokal-sneath i"||method=="sokal-sneat i"||method=="sokal-snea i"||method=="sokal-sne i"||method=="sokal-sn i"||method=="sokal-s i"||method=="ssi"||method=="ss i"||method==5){
       for(i in 1:ncol(m)){
        for(j in 1:ncol(m)){
           dist.mat[i,j] <- 1-(table(m[,i],m[,j])[2,2]/(2*(table(m[,i],m[,j])[1,2]+table(m[,i],m[,j])[2,1])+table(m[,i],m[,j])[2,2]))
        }
       }
      }

     ######## The Sokal & Sneath II Distance ########
      if(method == "Sokal-Sneath 2"||method=="Sokal-Sneat 2"||method=="Sokal-Snea 2"||method=="Sokal-Sne 2"||method=="Sokal-Sn 2"||method=="Sokal-S 2"||method=="SS2"||method=="SS 2"||method=="Sokal-Sneath II"||method=="Sokal-Sneat II"||method=="Sokal-Snea II"||method=="Sokal-Sne II"||method=="Sokal-Sn II"||method=="Sokal-S II"||method=="SSII"||method=="SS II"||method == "sokal-sneath 2"||method=="sokal-sneat 2"||method=="sokal-snea 2"||method=="sokal-sne 2"||method=="sokal-sn 2"||method=="sokal-s 2"||method=="ss2"||method=="ss 2"||method=="sokal-sneath ii"||method=="sokal-sneat ii"||method=="sokal-snea ii"||method=="sokal-sne ii"||method=="sokal-sn ii"||method=="sokal-s ii"||method=="ssii"||method=="ss ii"||method==6){
       for(i in 1:ncol(m)){
        for(j in 1:ncol(m)){
           dist.mat[i,j] <- 1-((2*(table(m[,i],m[,j])[2,2]+table(m[,i],m[,j])[1,1]))/(table(m[,i],m[,j])[1,2]+table(m[,i],m[,j])[2,1]+2*(table(m[,i],m[,j])[2,2]+table(m[,i],m[,j])[1,1])))
        }
       }
      }

     ######## The Sokal & Sneath III Distance ########
      if(method == "Sokal-Sneath 3"||method=="Sokal-Sneat 3"||method=="Sokal-Snea 3"||method=="Sokal-Sne 3"||method=="Sokal-Sn 3"||method=="Sokal-S 3"||method=="SS3"||method=="SS 3"||method=="Sokal-Sneath III"||method=="Sokal-Sneat III"||method=="Sokal-Snea III"||method=="Sokal-Sne III"||method=="Sokal-Sn III"||method=="Sokal-S III"||method=="SSIII"||method=="SS III"||method == "sokal-sneath 3"||method=="sokal-sneat 3"||method=="sokal-snea 3"||method=="sokal-sne 3"||method=="sokal-sn 3"||method=="sokal-s 3"||method=="ss3"||method=="ss 3"||method=="sokal-sneath iii"||method=="sokal-sneat iii"||method=="sokal-snea iii"||method=="sokal-sne iii"||method=="sokal-sn iii"||method=="sokal-s iii"||method=="ssiii"||method=="ss iii"||method==7){
       for(i in 1:ncol(m)){
        for(j in 1:ncol(m)){
           dist.mat[i,j] <- 1-((table(m[,i],m[,j])[2,2]+table(m[,i],m[,j])[1,1])/(table(m[,i],m[,j])[1,2]+table(m[,i],m[,j])[2,1]))
        }
       }
      }

     ######## The Sokal & Sneath IV Distance ########
      if(method == "Sokal-Sneath 4"||method=="Sokal-Sneat 4"||method=="Sokal-Snea 4"||method=="Sokal-Sne 4"||method=="Sokal-Sn 4"||method=="Sokal-S 4"||method=="SS4"||method=="SS 4"||method=="Sokal-Sneath IV"||method=="Sokal-Sneat IV"||method=="Sokal-Snea IV"||method=="Sokal-Sne IV"||method=="Sokal-Sn IV"||method=="Sokal-S IV"||method=="SSIV"||method=="SS IV"||method == "sokal-sneath 4"||method=="sokal-sneat 4"||method=="sokal-snea 4"||method=="sokal-sne 4"||method=="sokal-sn 4"||method=="sokal-s 4"||method=="ss4"||method=="ss 4"||method=="sokal-sneath iv"||method=="sokal-sneat iv"||method=="sokal-snea iv"||method=="sokal-sne iv"||method=="sokal-sn iv"||method=="sokal-s iv"||method=="ssiv"||method=="ss iv"||method==8){
       for(i in 1:ncol(m)){
        for(j in 1:ncol(m)){
           dist.mat[i,j] <- 1-(0.25*(table(m[,i],m[,j])[2,2]/(table(m[,i],m[,j])[2,2]+table(m[,i],m[,j])[2,1])+table(m[,i],m[,j])[2,2]/(table(m[,i],m[,j])[2,2]+table(m[,i],m[,j])[1,2])+table(m[,i],m[,j])[1,1]/(table(m[,i],m[,j])[1,1]+table(m[,i],m[,j])[2,1])+table(m[,i],m[,j])[1,1]/(table(m[,i],m[,j])[1,1]+table(m[,i],m[,j])[1,2])))
        }
       }
      }

     ######## The Sokal & Sneath V (or Ochiai 2) Distance ########
      if(method == "Sokal-Sneath 5"||method=="Sokal-Sneat 5"||method=="Sokal-Snea 5"||method=="Sokal-Sne 5"||method=="Sokal-Sn 5"||method=="Sokal-S 5"||method=="SS5"||method=="SS 5"||method=="Sokal-Sneath V"||method=="Sokal-Sneat V"||method=="Sokal-Snea V"||method=="Sokal-Sne V"||method=="Sokal-Sn V"||method=="Sokal-S V"||method=="SSv"||method=="SS V"||method=="Ochiai II"||method=="Ochia II"||method=="Ochi II"||method=="Och II"||method=="Oc II"||method=="O II"||method=="O2"||method=="O 2"||method=="OII"||method=="O II"||method == "sokal-sneath 5"||method=="sokal-sneat 5"||method=="sokal-snea 5"||method=="sokal-sne 5"||method=="sokal-sn 5"||method=="sokal-s 5"||method=="ss5"||method=="ss 5"||method=="sokal-sneath v"||method=="sokal-sneat v"||method=="sokal-snea v"||method=="sokal-sne v"||method=="sokal-sn v"||method=="sokal-s v"||method=="ssv"||method=="ss v"||method=="ochiai ii"||method=="ochia ii"||method=="ochi ii"||method=="och ii"||method=="oc ii"||method=="o ii"||method=="o2"||method=="o 2"||method=="oii"||method=="o ii"||method==9){
       for(i in 1:ncol(m)){
        for(j in 1:ncol(m)){
           dist.mat[i,j] <- 1-((table(m[,i],m[,j])[2,2]*table(m[,i],m[,j])[1,1])/(sqrt(((table(m[,i],m[,j])[2,2]+table(m[,i],m[,j])[2,1])*(table(m[,i],m[,j])[2,2]+table(m[,i],m[,j])[1,2])*(table(m[,i],m[,j])[1,1]+table(m[,i],m[,j])[2,1])*(table(m[,i],m[,j])[1,1]+table(m[,i],m[,j])[1,2])))))
        }
       }
      }

     ######## The Rogers & Tanimoto Distance ########
      if(method == "Rogers-Tanimoto"||method=="Rogers-Tanimot"||method=="Rogers-Tanimo"||method=="Rogers-Tanim"||method=="Rogers-Tani"||method=="Rogers-Tan"||method=="Rogers-Ta"||method=="Rogers-T"||method=="Rogers-"||method=="Roger"||method=="Roge"||method=="Rog"||method=="RT"||method == "rogers-tanimoto"||method=="rogers-tanimot"||method=="rogers-tanimo"||method=="rogers-tanim"||method=="rogers-tani"||method=="rogers-tan"||method=="rogers-ta"||method=="rogers-t"||method=="rogers-"||method=="roger"||method=="roge"||method=="rog"||method=="rt"||method==10){
       for(i in 1:ncol(m)){
        for(j in 1:ncol(m)){
           dist.mat[i,j] <- 1-((table(m[,i],m[,j])[2,2]+table(m[,i],m[,j])[1,1])/(2*(table(m[,i],m[,j])[1,2]+table(m[,i],m[,j])[2,1])+table(m[,i],m[,j])[2,2]+table(m[,i],m[,j])[1,1]))
        }
       }
      }

     ######## The Faith Distance ########
      if(method == "Faith"||method=="Fait"||method=="Fai"||method=="Fa"||method=="F"||method == "faith"||method=="fait"||method=="fai"||method=="fa"||method=="f"||method==11){
       for(i in 1:ncol(m)){
        for(j in 1:ncol(m)){
           dist.mat[i,j] <- 1-((table(m[,i],m[,j])[2,2]+0.5*table(m[,i],m[,j])[1,1])/(table(m[,i],m[,j])[1,2]+table(m[,i],m[,j])[2,1]+table(m[,i],m[,j])[2,2]+table(m[,i],m[,j])[1,1]))
        }
       }
      }

     ######## The Gower & Legendre Distance ########
      if(method == "Gower-Legendre"||method=="Gower-Legendr"||method=="Gower-Legend"||method=="Gower-Legen"||method=="Gower-Lege"||method=="Gower-Leg"||method=="Gower-Le"||method=="Gower-L"||method=="Gower-"||method=="GL"||method == "gower-legendre"||method=="gower-legendr"||method=="gower-legend"||method=="gower-legen"||method=="gower-lege"||method=="gower-leg"||method=="gower-le"||method=="gower-l"||method=="gower-"||method=="gl"||method==12){
       for(i in 1:ncol(m)){
        for(j in 1:ncol(m)){
           dist.mat[i,j] <- 1-((table(m[,i],m[,j])[2,2]+table(m[,i],m[,j])[1,1])/(0.5*(table(m[,i],m[,j])[1,2]+table(m[,i],m[,j])[2,1])+table(m[,i],m[,j])[2,2]+table(m[,i],m[,j])[1,1]))
        }
       }
      }

     ######## The Russell & Rao Distance ########
      if(method == "Russell-Rao"||method=="Russell-Ra"||method=="Russell-R"||method=="Russell-"||method=="Russell"||method=="Russel"||method=="Russe"||method=="Russ"||method=="Rus"||method=="RR"||method == "russell-rao"||method=="russell-ra"||method=="russell-r"||method=="russell-"||method=="russell"||method=="russel"||method=="russe"||method=="russ"||method=="rus"||method=="rr"||method==13){
       for(i in 1:ncol(m)){
        for(j in 1:ncol(m)){
           dist.mat[i,j] <- 1-((table(m[,i],m[,j])[2,2])/(table(m[,i],m[,j])[1,2]+table(m[,i],m[,j])[2,1]+table(m[,i],m[,j])[2,2]+table(m[,i],m[,j])[1,1]))
        }
       }
      }

     ######## The Ochiai I (or Otsuka & Ochiai, or Cosine, or Fowlkes and Mallows Index) Distance ########
      if(method == "Ochiai I"||method == "Ochia I"||method == "Ochi I"||method == "Och I"||method == "Oc I"||method == "O I"||method == "Ochiai 1"||method == "Ochia 1"||method == "Ochi 1"||method == "Och 1"||method == "Oc 1"||method == "O 1"||method == "OI"||method == "O1"||method=="ochiai i"||method=="ochia i"||method=="ochi i"||method=="och i"||method=="oc i"||method=="ochiai 1"||method=="ochia 1"||method=="ochi 1"||method=="och 1"||method=="oc 1"||method=="o1"||method=="o 1"||method=="oi"||method=="o i"||method == "Cosine"||method == "Cosin"||method == "Cosi"||method == "Cos"||method=="cosine"||method=="cosin"||method=="cosi"||method=="cos"||method == "Otsuka-Ochiai"||method == "Otsuka-Ochia"||method == "Otsuka-Ochi"||method == "Otsuka-Och"||method == "Otsuka-Oc"||method == "Otsuka-O"||method == "Otsuka-"||method == "Otsuka"||method == "Otsuk"||method == "Otsuk"||method == "Otsu"||method == "Ots"||method=="otsuka-ochiai"||method=="otsuka-ochia"||method=="otsuka-ochi"||method=="otsuka-och"||method=="otsuka-oc"||method=="otsuka-o"||method=="otsuka-"||method=="otsuka"||method=="otsuk"||method=="otsu"||method=="ots"||method=="oo"||method=="OO"||method == "Fowlkes-Mallows"||method == "Fowlkes-Mallow"||method == "Fowlkes-Mallo"||method == "Fowlkes-Mall"||method == "Fowlkes-Mal"||method == "Fowlkes-Ma"||method == "Fowlkes-M"||method == "Fowlkes-"||method == "Fowlkes"||method == "Fowlke"||method == "Fowlk"||method == "Fowl"||method=="fowlkes-mallows"||method=="fowlkes-mallow"||method=="fowlkes-mallo"||method=="fowlkes-mall"||method=="fowlkes-mal"||method=="fowlkes-ma"||method=="fowlkes-m"||method=="fowlkes-"||method=="fowlkes"||method=="fowlke"||method=="fowlk"||method=="fowl"||method=="FM"||method=="fm"||method==14){
       for(i in 1:ncol(m)){
        for(j in 1:ncol(m)){
           dist.mat[i,j] <- 1-((table(m[,i],m[,j])[2,2])/(sqrt((table(m[,i],m[,j])[2,2]+table(m[,i],m[,j])[2,1])*(table(m[,i],m[,j])[2,2]+table(m[,i],m[,j])[1,2]))))
        }
       }
      }

     ######## The Forbes I Distance ########
      if(method=="Forbes I"||method=="Forbe I"||method=="Forb I"||method=="For I"||method=="Forbes 1"||method=="Forbe 1"||method=="Forb 1"||method=="For 1"||method=="F1"||method=="FI"||method=="F 1"||method=="F I"||method=="forbes i"||method=="forbe i"||method=="forb i"||method=="for i"||method=="forbes 1"||method=="forbe 1"||method=="forb 1"||method=="for 1"||method=="f1"||method=="fi"||method=="f 1"||method=="f i"||method==15){
       for(i in 1:ncol(m)){
        for(j in 1:ncol(m)){
           dist.mat[i,j] <- 1-(((table(m[,i],m[,j])[2,2]+table(m[,i],m[,j])[2,1]+table(m[,i],m[,j])[1,2]+table(m[,i],m[,j])[1,1])*(table(m[,i],m[,j])[2,2]))/((table(m[,i],m[,j])[2,2]+table(m[,i],m[,j])[2,1])*(table(m[,i],m[,j])[2,2]+table(m[,i],m[,j])[1,2])))
        }
       }
